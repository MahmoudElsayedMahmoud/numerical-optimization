theta_0=[]
theta_1=[]
alpha=0.001
ress=[]
theta0=0
theta1=0

for i in range(100):
    h=theta0+theta1*x
    error = sum((h-y)**2)
    res = error/ (2 * len(x))
    ress.append(res)
    a = sum(h-y) / ( len(x))
    b = sum((h-y)*x) / ( len(x))
    theta0 = theta0 - alpha*a
    theta_0.append(theta0)
    theta1 = theta1 - alpha*b 
    theta_1.append(theta1)

theta_0=[]
theta_1=[]
theta_2=[]
theta_3=[]
alpha=0.005
ress=[]
theta0=0
theta1=0
theta2=0
theta3=0

for i in range(50):
    h=theta0*x[:,0:1]+theta1*x[:,1:2]+theta2*x[:,2:3]+theta3*x[:,3:4]
    error = sum((h-y)**2)
    res = error/ (2 * len(x))
    ress.append(res)
    a = sum(h-y)*x[:,0:1] / ( len(x[:,0:1]))
    b = sum((h-y)*x[:,1:2]) / ( len(x[:,1:2]))
    c = sum((h-y)*x[:,2:3]) / ( len(x[:,2:3]))
    d = sum((h-y)*x[:,3:4]) / ( len(x[:,3:4]))
    theta0 = theta0 - alpha*a
    theta_0.append(theta0)
    theta1 = theta1 - alpha*b 
    theta_1.append(theta1)
    theta1 = theta2 - alpha*c 
    theta_2.append(theta2)
    theta1 = theta3 - alpha*d 
    theta_3.append(theta3)
